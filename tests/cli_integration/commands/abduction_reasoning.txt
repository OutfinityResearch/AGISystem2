# Test: Abductive reasoning and counterfactuals
# Tests: abduct, cf, hypothesize

# Setup causal facts
add Fire CAUSES Smoke
add Smoke CAUSED_BY Fire
add Rain CAUSES WetGround
add WetGround CAUSED_BY Rain
add Sprinkler CAUSES WetGround

# Abductive reasoning - find causes
abduct Smoke CAUSES
abduct WetGround CAUSED_BY

# Setup for counterfactual
add Earth HAS_PROPERTY gravity=9.8

# Counterfactual query
cf Earth HAS_PROPERTY gravity=0? | Earth HAS_PROPERTY gravity=0

# Generate hypotheses
hypothesize Dog
hypothesize Fire
